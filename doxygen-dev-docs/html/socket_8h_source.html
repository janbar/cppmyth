412  size_t ReceiveData(void* buf, size_t n)
413  {
414  size_t r = 0;
415  if (m_bound.IsValid())
416  {
417  if ((r = m_bound.ReadData(buf, n)) > 0)
418  return r;
419  if (m_bound.AwaitIncoming(m_timeout) > 0)
420  return m_bound.ReadData(buf, n);
421  }
422  return r;
423  }
