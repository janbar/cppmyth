135 bool LiveTVPlayback::SpawnLiveTV(const std::string& chanNum, const ChannelList& channels)
136 {
137  // Begin critical section
138  OS::CLockGuard lock(*m_mutex);
139  if (!ProtoMonitor::IsOpen() || !m_eventHandler.IsConnected())
140  {
141  DBG(MYTH_DBG_ERROR, "%s: not connected\n", __FUNCTION__);
142  return false;
143  }
144 
145  StopLiveTV();
146  preferredCards_t preferredCards = FindTunableCardIds(chanNum, channels);
147  preferredCards_t::const_iterator card = preferredCards.begin();
148  while (card != preferredCards.end())
149  {
150  InitChain(); // Setup chain
151  const CardInputPtr& input = card->second.first;
152  const ChannelPtr& channel = card->second.second;
153  DBG(MYTH_DBG_DEBUG, "%s: trying recorder num (%" PRIu32 ") channum (%s)\n", __FUNCTION__, input->cardId, channel->chanNum.c_str());
154  m_recorder = GetRecorderFromNum((int) input->cardId);
155  // Setup the chain
156  m_chain.switchOnCreate = true;
157  m_chain.watch = true;
158  if (m_recorder->SpawnLiveTV(m_chain.UID, channel->chanNum))
159  {
160  // Wait chain update until time limit
161  uint32_t delayMs = m_tuneDelay * 1000;
162  OS::CTimeout timeout(delayMs);
163  do
164  {
165  lock.Unlock(); // Release the latch to allow chain update
166  usleep(TICK_USEC);
167  lock.Lock();
168  if (!m_chain.switchOnCreate)
169  {
170  DBG(MYTH_DBG_DEBUG, "%s: tune delay (%" PRIu32 "ms)\n", __FUNCTION__, (delayMs - timeout.TimeLeft()));
171  return true;
172  }
173  }
174  while (timeout.TimeLeft() > 0);
175  DBG(MYTH_DBG_ERROR, "%s: tune delay exceeded (%" PRIu32 "ms)\n", __FUNCTION__, delayMs);
176  m_recorder->StopLiveTV();
177  }
178  ClearChain();
179  // Retry the next preferred card
180  ++card;
181  }
182  return false;
183 }

244 void LiveTVPlayback::HandleChainUpdate()
245 {
246  OS::CLockGuard lock(*m_mutex); // Lock chain
247  ProtoRecorderPtr recorder(m_recorder);
248  if (!recorder)
249  return;
250  ProgramPtr prog = recorder->GetCurrentRecording();
251  /*
252  * If program file doesn't exist in the recorder chain then create a new
253  * transfer and add it to the chain.
254  */
255  if (prog && !prog->fileName.empty() && !IsChained(*prog))
256  {
257  DBG(MYTH_DBG_DEBUG, "%s: liveTV (%s): adding new transfer %s\n", __FUNCTION__,
258  m_chain.UID.c_str(), prog->fileName.c_str());
259  ProtoTransferPtr transfer(new ProtoTransfer(recorder->GetServer(), recorder->GetPort(), prog->fileName, prog->recording.storageGroup));
260  // Pop previous dummy file if exists then add the new into the chain
261  if (m_chain.lastSequence && m_chain.chained[m_chain.lastSequence - 1].first->GetSize() == 0)
262  {
263  --m_chain.lastSequence;
264  m_chain.chained.pop_back();
265  }
266  m_chain.chained.push_back(std::make_pair(transfer, prog));
267  m_chain.lastSequence = m_chain.chained.size();
268  /*
269  * If switchOnCreate flag and file is filled then switch immediatly.
270  * Else we will switch later on the next event 'UPDATE_FILE_SIZE'
271  */
272  if (m_chain.switchOnCreate && transfer->GetSize() > 0 && SwitchChainLast())
273  m_chain.switchOnCreate = false;
274  m_chain.watch = false; // Chain update done. Restore watch flag
275  DBG(MYTH_DBG_DEBUG, "%s: liveTV (%s): chain last (%u), watching (%u)\n", __FUNCTION__,
276  m_chain.UID.c_str(), m_chain.lastSequence, m_chain.currentSequence);
277  }
278 }

280 bool LiveTVPlayback::SwitchChain(unsigned sequence)
281 {
282  OS::CLockGuard lock(*m_mutex);
283  // Check for out of range
284  if (sequence < 1 || sequence > m_chain.lastSequence)
285  return false;
286  // If closed then try to open
287  if (!m_chain.chained[sequence - 1].first->IsOpen() && !m_chain.chained[sequence - 1].first->Open())
288  return false;
289  m_chain.currentTransfer = m_chain.chained[sequence - 1].first;
290  m_chain.currentSequence = sequence;
291  DBG(MYTH_DBG_DEBUG, "%s: switch to file (%u) %s\n", __FUNCTION__,
292  (unsigned)m_chain.currentTransfer->GetFileId(), m_chain.currentTransfer->GetPathName().c_str());
293  return true;
294 }

308 void LiveTVPlayback::HandleBackendMessage(EventMessagePtr msg)
309 {
310  ProtoRecorderPtr recorder(m_recorder);
311  if (!recorder || !recorder->IsPlaying())
312  return;
313  switch (msg->event)
314  {
315  /*
316  * Event: LIVETV_CHAIN UPDATE
317  *
318  * Called in response to the backend's notification of a chain update.
319  * The recorder is supplied and will be queried for the current recording
320  * to determine if a new file needs to be added to the chain of files
321  * in the live tv instance.
322  */
323  case EVENT_LIVETV_CHAIN:
324  if (msg->subject.size() >= 3)
325  {
326  if (msg->subject[1] == "UPDATE" && msg->subject[2] == m_chain.UID)
327  HandleChainUpdate();
328  }
329  break;
330  /*
331  * Event: LIVETV_WATCH
332  *
333  * Called in response to the backend's notification of a livetv watch.
334  * The recorder is supplied and will be updated for the watch signal.
335  * This event is used to manage program breaks while watching live tv.
336  * When the guide data marks the end of one show and the beginning of
337  * the next, which will be recorded to a new file, this instructs the
338  * frontend to terminate the existing playback, and change channel to
339  * the new file. Before updating livetv chain and switching to new file
340  * we must to wait for event DONE_RECORDING that informs the current
341  * show is completed. Then we will call livetv chain update to get
342  * current program info. Watch signal will be down during this period.
343  */
344  case EVENT_LIVETV_WATCH:
345  if (msg->subject.size() >= 3)
346  {
347  int32_t rnum;
348  int8_t flag;
349  if (string_to_int32(msg->subject[1].c_str(), &rnum) == 0 && string_to_int8(msg->subject[2].c_str(), &flag) == 0)
350  {
351  if (recorder->GetNum() == (int)rnum)
352  {
353  OS::CLockGuard lock(*m_mutex); // Lock chain
354  m_chain.watch = true;
355  }
356  }
357  }
358  break;
359  /*
360  * Event: DONE_RECORDING
361  *
362  * Indicates that an active recording has completed on the specified
363  * recorder. used to manage program breaks while watching live tv.
364  * When receive event for recorder, we force an update of livetv chain
365  * to get current program info when chain is not yet updated.
366  * Watch signal is used when up, to mark the break period and
367  * queuing the frontend for reading file buffer.
368  */
369  case EVENT_DONE_RECORDING:
370  if (msg->subject.size() >= 2)
371  {
372  int32_t rnum;
373  if (string_to_int32(msg->subject[1].c_str(), &rnum) == 0 && recorder->GetNum() == (int)rnum)
374  {
375  // Recorder is not subscriber. So callback event to it
376  recorder->DoneRecordingCallback();
377  // Manage program break
378  if (m_chain.watch)
379  {
380  /*
381  * Last recording is now completed but watch signal is ON.
382  * Then force live tv chain update for the new current
383  * program. We will retry for 2s before returning.
384  */
385  OS::CTimeout timeout(2000);
386  do
387  {
388  usleep(500000); // wait for 500 ms
389  HandleChainUpdate();
390  }
391  while (m_chain.watch && timeout.TimeLeft() > 0);
392  }
393  }
394  }
395  break;
396  case EVENT_UPDATE_FILE_SIZE:
397  if (msg->subject.size() >= 3)
398  {
399  OS::CLockGuard lock(*m_mutex); // Lock chain
400  if (m_chain.lastSequence > 0)
401  {
402  int64_t newsize;
403  // Message contains chanid + starttime as recorded key
404  if (msg->subject.size() >= 4)
405  {
406  uint32_t chanid;
407  time_t startts;
408  if (string_to_uint32(msg->subject[1].c_str(), &chanid)
409  || string_to_time(msg->subject[2].c_str(), &startts)
410  || m_chain.chained[m_chain.lastSequence -1].second->channel.chanId != chanid
411  || m_chain.chained[m_chain.lastSequence -1].second->recording.startTs != startts
412  || string_to_int64(msg->subject[3].c_str(), &newsize)
413  || m_chain.chained[m_chain.lastSequence - 1].first->GetSize() >= newsize)
414  break;
415  }
416  // Message contains recordedid as key
417  else
418  {
419  uint32_t recordedid;
420  if (string_to_uint32(msg->subject[1].c_str(), &recordedid)
421  || m_chain.chained[m_chain.lastSequence -1].second->recording.recordedId != recordedid
422  || string_to_int64(msg->subject[2].c_str(), &newsize)
423  || m_chain.chained[m_chain.lastSequence - 1].first->GetSize() >= newsize)
424  break;
425  }
426  // Update transfer file size
427  m_chain.chained[m_chain.lastSequence - 1].first->SetSize(newsize);
428  // Is wait the filling before switching ?
429  if (m_chain.switchOnCreate && SwitchChainLast())
430  m_chain.switchOnCreate = false;
431  DBG(MYTH_DBG_DEBUG, "%s: liveTV (%s): chain last (%u) filesize %" PRIi64 "\n", __FUNCTION__,
432  m_chain.UID.c_str(), m_chain.lastSequence, newsize);
433  }
434  }
435  break;
436  case EVENT_SIGNAL:
437  if (msg->subject.size() >= 2)
438  {
439  int32_t rnum;
440  if (string_to_int32(msg->subject[1].c_str(), &rnum) == 0 && recorder->GetNum() == (int)rnum)
441  m_signal = msg->signal;
442  }
443  break;
444  //case EVENT_HANDLER_STATUS:
445  // if (msg->subject[0] == EVENTHANDLER_DISCONNECTED)
446  // closeTransfer();
447  // break;
448  default:
449  break;
450  }
451 }

462 int LiveTVPlayback::Read(void* buffer, unsigned n)
463 {
464  int r = 0;
465  bool retry;
466  int64_t s, fp;
467 
468  // Begin critical section
469  // First of all i hold my shared resources using copies
470  ProtoRecorderPtr recorder(m_recorder);
471  if (!m_chain.currentTransfer || !recorder)
472  return -1;
473 
474  fp = m_chain.currentTransfer->GetPosition();
475 
476  do
477  {
478  retry = false;
479  s = m_chain.currentTransfer->GetRemaining(); // Acceptable block size
480  if (s == 0)
481  {
482  OS::CTimeout timeout(AHEAD_TIMEOUT);
483  for (;;)
484  {
485  // Reading ahead
486  if (m_chain.currentSequence == m_chain.lastSequence)
487  {
488  int64_t rp = recorder->GetFilePosition();
489  if (rp > fp)
490  {
491  m_chain.currentTransfer->SetSize(rp);
492  retry = true;
493  break;
494  }
495  if (!timeout.TimeLeft())
496  {
497  DBG(MYTH_DBG_WARN, "%s: read position is ahead (%" PRIi64 ")\n", __FUNCTION__, fp);
498  return 0;
499  }
500  usleep(500000);
501  }
502  // Switch next file transfer is required to continue
503  else
504  {
505  if (!SwitchChain(m_chain.currentSequence + 1))
506  return -1;
507  if (m_chain.currentTransfer->GetPosition() != 0)
508  recorder->TransferSeek(*(m_chain.currentTransfer), 0, WHENCE_SET);
509  DBG(MYTH_DBG_DEBUG, "%s: liveTV (%s): chain last (%u), watching (%u)\n", __FUNCTION__,
510  m_chain.UID.c_str(), m_chain.lastSequence, m_chain.currentSequence);
511  retry = true;
512  break;
513  }
514  }
515  }
516  else if (s < 0)
517  return -1;
518  }
519  while (retry);
520 
521  if (s < (int64_t)n)
522  n = (unsigned)s ;
523 
524  r = recorder->TransferRequestBlock(*(m_chain.currentTransfer), buffer, n);
525  return r;
526 }

528 int64_t LiveTVPlayback::Seek(int64_t offset, WHENCE_t whence)
529 {
530  OS::CLockGuard lock(*m_mutex); // Lock chain
531  if (!m_recorder || !m_chain.currentSequence)
532  return -1;
533 
534  unsigned ci = m_chain.currentSequence - 1; // current sequence index
535  int64_t size = GetSize(); // total stream size
536  int64_t position = GetPosition(); // absolute position in stream
537  int64_t p = 0;
538  switch (whence)
539  {
540  case WHENCE_SET:
541  p = offset;
542  break;
543  case WHENCE_END:
544  p = size + offset;
545  break;
546  case WHENCE_CUR:
547  p = position + offset;
548  break;
549  default:
550  return -1;
551  }
552  if (p > size || p < 0)
553  {
554  DBG(MYTH_DBG_WARN, "%s: invalid seek (%" PRId64 ")\n", __FUNCTION__, p);
555  return -1;
556  }
557  if (p > position)
558  {
559  for (;;)
560  {
561  if (position + m_chain.chained[ci].first->GetRemaining() >= p)
562  {
563  // Try seek file to desired position. On success switch chain
564  if (m_recorder->TransferSeek(*(m_chain.chained[ci].first), p - position, WHENCE_CUR) < 0 ||
565  !SwitchChain(++ci))
566  return -1;
567  return p;
568  }
569  position += m_chain.chained[ci].first->GetRemaining();
570  ++ci; // switch next
571  if (ci < m_chain.lastSequence)
572  position += m_chain.chained[ci].first->GetPosition();
573  else
574  return -1;
575  }
576  }
577  if (p < position)
578  {
579  for (;;)
580  {
581  if (position - m_chain.chained[ci].first->GetPosition() <= p)
582  {
583  // Try seek file to desired position. On success switch chain
584  if (m_recorder->TransferSeek(*(m_chain.chained[ci].first), p - position, WHENCE_CUR) < 0 ||
585  !SwitchChain(++ci))
586  return -1;
587  return p;
588  }
589  position -= m_chain.chained[ci].first->GetPosition();
590  if (ci > 0)
591  {
592  --ci; // switch previous
593  position -= m_chain.chained[ci].first->GetRemaining();
594  }
595  else
596  return -1;
597  }
598  }
599  // p == position
600  return p;
601 }

698 LiveTVPlayback::preferredCards_t LiveTVPlayback::FindTunableCardIds(const std::string& chanNum, const ChannelList& channels)
699 {
700  // Make the set of channels matching the desired channel number
701  ChannelList chanset;
702  for (ChannelList::const_iterator it = channels.begin(); it != channels.end(); ++it)
703  {
704  if ((*it)->chanNum == chanNum)
705  chanset.push_back(*it);
706  }
707  // Retrieve unlocked encoders and fill the list of preferred cards.
708  // It is ordered by its key liveTVOrder and contains matching between channels
709  // and card inputs using their respective sourceId and mplexId
710  std::vector<int> ids = GetFreeCardIdList();
711  preferredCards_t preferredCards;
712  for (std::vector<int>::const_iterator itc = ids.begin(); itc != ids.end(); ++itc)
713  {
714  CardInputListPtr inputs = GetFreeInputs(*itc);
715  for (CardInputList::const_iterator iti = inputs->begin(); iti != inputs->end(); ++iti)
716  {
717  for (ChannelList::const_iterator itchan = chanset.begin(); itchan != chanset.end(); ++itchan)
718  {
719  if ((*itchan)->sourceId == (*iti)->sourceId && ( (*iti)->mplexId == 0 || (*iti)->mplexId == (*itchan)->mplexId ))
720  {
721  preferredCards.insert(std::make_pair((*iti)->liveTVOrder, std::make_pair(*iti, *itchan)));
722  DBG(MYTH_DBG_DEBUG, "%s: [%u] channel=%s(%" PRIu32 ") card=%" PRIu32 " input=%s(%" PRIu32 ") mplex=%" PRIu32 " source=%" PRIu32 "\n",
723  __FUNCTION__, (*iti)->liveTVOrder, (*itchan)->callSign.c_str(), (*itchan)->chanId,
724  (*iti)->cardId, (*iti)->inputName.c_str(), (*iti)->inputId, (*iti)->mplexId, (*iti)->sourceId);

726  }
727  }
728  }
729  }
