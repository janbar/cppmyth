62 static inline void __dbg(debug_ctx_t *ctx, int level, const char *fmt, va_list ap)
63 {
64  char msg[4096];
65  int len;
66  if (ctx == NULL) {
67  return;
68  }
69  if (level <= ctx->cur_level) {
70  len = snprintf(msg, sizeof(msg), "(%s)", ctx->name);
71  vsnprintf(msg + len, sizeof(msg)-len, fmt, ap);
72  if (ctx->msg_callback) {
73  ctx->msg_callback(level, msg);
74  } else {
75  fwrite(msg, strlen(msg), 1, stderr);
76  }
77  }
78 }
