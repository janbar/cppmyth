87 bool Control::RefreshRecordedArtwork(Program& program)
88 {
89  program.artwork.clear();
90  if (program.inetref.empty())
91  return false;
92  ArtworkListPtr artworks(GetRecordingArtworkList(program.channel.chanId, program.recording.startTs));
93  program.artwork.reserve(artworks->size());
94  for (ArtworkList::const_iterator it = artworks->begin(); it < artworks->end(); ++it)
95  {
96  program.artwork.push_back(*(it->get()));
97  }
98  return (program.artwork.empty() ? false : true);
