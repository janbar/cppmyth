30 static void __form_urlencode(std::string& encoded, const char *str)
31 {
32  char buf[4];
33  size_t i, len = 0;
34 
35  encoded.clear();
36  if (str != NULL)
37  len = strlen(str);
38  encoded.reserve(len * 3);
39 
40  for (i = 0; i < len; ++i)
41  {
42  sprintf(buf, "%%%.2X", (unsigned char)str[i]);
43  encoded.append(buf);
44  }
45 }

74 void WSRequest::SetContentParam(const std::string& param, const std::string& value)
75 {

97 void WSRequest::MakeMessageGET(std::string& msg) const
98 {
99  char buf[32];
100 
101  msg.clear();
102  msg.reserve(256);
103  msg.append("GET ").append(m_service_url);
104  if (!m_contentData.empty())
105  msg.append("?").append(m_contentData);
106  msg.append(" " REQUEST_PROTOCOL "\r\n");
107  sprintf(buf, "%u", m_port);
108  msg.append("Host: ").append(m_server).append(":").append(buf).append("\r\n");
109  msg.append("User-Agent: " REQUEST_USER_AGENT "\r\n");
110  msg.append("Connection: " REQUEST_CONNECTION "\r\n");
111  if (m_accept != CT_NONE)
112  msg.append("Accept: ").append(MimeFromContentType(m_accept)).append("\r\n");
113  msg.append("Accept-Charset: ").append(m_charset).append("\r\n");
114  msg.append("\r\n");
115 }

126  msg.append("Host: ").append(m_server).append(":").append(buf).append("\r\n");
127  msg.append("User-Agent: " REQUEST_USER_AGENT "\r\n");
128  msg.append("Connection: " REQUEST_CONNECTION "\r\n");
129  if (m_accept != CT_NONE)
130  msg.append("Accept: ").append(MimeFromContentType(m_accept)).append("\r\n");
131  msg.append("Accept-Charset: ").append(m_charset).append("\r\n");
132  if (content_len)
133  {
134  sprintf(buf, "%lu", (unsigned long)content_len);
135  msg.append("Content-Type: ").append(MimeFromContentType(m_contentType));
136  msg.append("; charset=" REQUEST_STD_CHARSET "\r\n");
137  msg.append("Content-Length: ").append(buf).append("\r\n\r\n");
138  msg.append(m_contentData);
139  }
140  else
141  msg.append("\r\n");
142 }
