100 int __tValFromString(protoref_t *map, unsigned sz, unsigned proto, const std::string& sVal, int unk)
101 {
102  for (unsigned i = 0; i < sz; ++i)
103  {
104  if (proto >= map[i].protoVer && sVal.compare(map[i].sVal) == 0)
105  return map[i].tVal;
106  }
107  return unk;
108 }
